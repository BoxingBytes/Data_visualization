library(shiny)
library(dplyr)
library(ggplot2)
library(arrow)
library(sf)
library(leaflet)
library(lubridate)
library(scales)
library(DT)

# Load data
validations <- read_parquet("data/validations_sample.parquet")
zd <- st_read("data/zd.shp", quiet = TRUE)

# Prepare zd lookup
zd <- zd %>%
  mutate(
    idrefa_lda = as.character(idrefa_lda),
    idrefa_lda = sub("\\.0$", "", idrefa_lda)
  )

zd_lookup <- zd %>%
  st_drop_geometry() %>%
  select(idrefa_lda, type_arret, nom_lda) %>%
  distinct(idrefa_lda, .keep_all = TRUE)

# Join validations with station info
validations <- validations %>%
  left_join(zd_lookup, by = c("ID_REFA_LDA" = "idrefa_lda"))

# Check and transform CRS for leaflet - was a bug for openstreetmap display
if(!is.na(st_crs(zd)) && st_crs(zd)$epsg != 4326) {
  zd_wgs84 <- st_transform(zd, 4326)
} else if(is.na(st_crs(zd))) {
  st_crs(zd) <- 2154  # Assume Lambert 93
  zd_wgs84 <- st_transform(zd, 4326)
} else {
  zd_wgs84 <- zd
}

# Calculate THE NORM (excluding holidays, COVID, incomplete weeks)
covid_start <- as.Date("2020-03-01")
covid_end <- as.Date("2022-06-30")

norm_data <- validations %>%
  filter(
    is_holiday == "Normal Day",  # Exclude holidays
    !(JOUR >= covid_start & JOUR <= covid_end)  # Exclude COVID period
  )

# Calculate norm by weekday
norm_by_weekday <- norm_data %>%
  group_by(JOUR, day_of_week) %>%
  summarise(daily_total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
  group_by(day_of_week) %>%
  summarise(
    norm_avg = mean(daily_total, na.rm = TRUE),
    norm_sd = sd(daily_total, na.rm = TRUE),
    .groups = "drop"
  )

# Server
server <- function(input, output, session) {
  
  # Update station choices
  station_choices <- setNames(
    zd_lookup$idrefa_lda,
    zd_lookup$nom_lda
  )
  updateSelectInput(session, "station_select", choices = station_choices)
  
  # Reactive data for norm comparison period
  norm_comp_data <- reactive({
    validations %>%
      filter(JOUR >= input$norm_dates[1] & JOUR <= input$norm_dates[2])
  })
  
  # Norm comparison table
  output$norm_comparison_table <- renderDT({
    selected <- norm_comp_data() %>%
      group_by(JOUR, day_of_week) %>%
      summarise(daily_total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      group_by(day_of_week) %>%
      summarise(avg_val = mean(daily_total, na.rm = TRUE), .groups = "drop")
    
    comparison <- norm_by_weekday %>%
      left_join(selected, by = "day_of_week") %>%
      mutate(
        deviation_pct = round((avg_val - norm_avg) / norm_avg * 100, 1),
        deviation_abs = round(avg_val - norm_avg, 0)
      ) %>%
      select(
        Weekday = day_of_week,
        `Norm Average` = norm_avg,
        `Selected Period` = avg_val,
        `Deviation (%)` = deviation_pct,
        `Deviation (abs)` = deviation_abs
      )
    
    datatable(comparison, options = list(dom = 't', ordering = FALSE), rownames = FALSE) %>%
      formatCurrency(c("Norm Average", "Selected Period", "Deviation (abs)"), 
                     currency = "", digits = 0) %>%
      formatStyle(
        'Deviation (%)',
        backgroundColor = styleInterval(c(-10, 10), c('#ffcccc', 'white', '#ccffcc'))
      )
  })
  
  # Norm weekday plot
  output$norm_weekday_plot <- renderPlot({
    ggplot(norm_by_weekday, aes(x = day_of_week, y = norm_avg)) +
      geom_col(fill = "darkblue", alpha = 0.7) +
      geom_errorbar(aes(ymin = norm_avg - norm_sd, ymax = norm_avg + norm_sd),
                    width = 0.2, color = "red") +
      scale_y_continuous(labels = comma) +
      labs(title = "THE NORM - Average Ridership by Weekday",
           subtitle = "Error bars show ± 1 standard deviation",
           x = "Day", y = "Average Validations") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
  
  # Selected period weekday plot
  output$selected_weekday_plot <- renderPlot({
    norm_comp_data() %>%
      group_by(JOUR, day_of_week) %>%
      summarise(daily_total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      group_by(day_of_week) %>%
      summarise(avg_val = mean(daily_total, na.rm = TRUE), .groups = "drop") %>%
      ggplot(aes(x = day_of_week, y = avg_val)) +
      geom_col(fill = "darkorange", alpha = 0.7) +
      scale_y_continuous(labels = comma) +
      labs(title = "Selected Period - Average by Weekday",
           x = "Day", y = "Average Validations") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
  
  # Deviation plot
  output$deviation_plot <- renderPlot({
    selected <- norm_comp_data() %>%
      group_by(JOUR, day_of_week) %>%
      summarise(daily_total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      group_by(day_of_week) %>%
      summarise(avg_val = mean(daily_total, na.rm = TRUE), .groups = "drop")
    
    norm_by_weekday %>%
      left_join(selected, by = "day_of_week") %>%
      mutate(deviation_pct = (avg_val - norm_avg) / norm_avg * 100) %>%
      ggplot(aes(x = day_of_week, y = deviation_pct, fill = deviation_pct > 0)) +
      geom_col(show.legend = FALSE) +
      geom_hline(yintercept = 0, linetype = "dashed") +
      scale_fill_manual(values = c("red", "green")) +
      labs(title = "Deviation from Norm (%)",
           x = "Day", y = "Deviation (%)") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
  
  # Reactive data for reference period
  ref_data <- reactive({
    validations %>%
      filter(JOUR >= input$ref_dates[1] & JOUR <= input$ref_dates[2])
  })
  
  # Reactive data for comparison period
  comp_data <- reactive({
    validations %>%
      filter(JOUR >= input$comp_dates[1] & JOUR <= input$comp_dates[2])
  })
  
  # Comparison table
  output$comparison_table <- renderDT({
    ref <- ref_data() %>%
      summarise(
        Total_Validations = sum(NB_VALD, na.rm = TRUE),
        Avg_Daily = mean(NB_VALD, na.rm = TRUE),
        Active_Stations = n_distinct(ID_REFA_LDA)
      ) %>%
      mutate(Period = "Reference")
    
    comp <- comp_data() %>%
      summarise(
        Total_Validations = sum(NB_VALD, na.rm = TRUE),
        Avg_Daily = mean(NB_VALD, na.rm = TRUE),
        Active_Stations = n_distinct(ID_REFA_LDA)
      ) %>%
      mutate(Period = "Comparison")
    
    diff <- data.frame(
      Period = "Difference (%)",
      Total_Validations = round((comp$Total_Validations - ref$Total_Validations) / ref$Total_Validations * 100, 2),
      Avg_Daily = round((comp$Avg_Daily - ref$Avg_Daily) / ref$Avg_Daily * 100, 2),
      Active_Stations = comp$Active_Stations - ref$Active_Stations
    )
    
    bind_rows(ref, comp, diff) %>%
      datatable(options = list(dom = 't', ordering = FALSE),
                rownames = FALSE) %>%
      formatCurrency(c("Total_Validations", "Avg_Daily"), currency = "", digits = 0)
  })
  
  # Reference weekday plot
  output$ref_weekday_plot <- renderPlot({
    ref_data() %>%
      group_by(JOUR, day_of_week) %>%
      summarise(daily_total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      group_by(day_of_week) %>%
      summarise(avg_val = mean(daily_total, na.rm = TRUE), .groups = "drop") %>%
      ggplot(aes(x = day_of_week, y = avg_val, fill = day_of_week)) +
      geom_col(show.legend = FALSE) +
      scale_y_continuous(labels = comma) +
      labs(title = "Reference Period - Avg by Weekday",
           x = "Day", y = "Avg Validations") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
  
  # Comparison weekday plot
  output$comp_weekday_plot <- renderPlot({
    comp_data() %>%
      group_by(JOUR, day_of_week) %>%
      summarise(daily_total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      group_by(day_of_week) %>%
      summarise(avg_val = mean(daily_total, na.rm = TRUE), .groups = "drop") %>%
      ggplot(aes(x = day_of_week, y = avg_val, fill = day_of_week)) +
      geom_col(show.legend = FALSE) +
      scale_y_continuous(labels = comma) +
      labs(title = "Comparison Period - Avg by Weekday",
           x = "Day", y = "Avg Validations") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
  
  # Daily comparison plot
  output$daily_comparison_plot <- renderPlot({
    ref <- ref_data() %>%
      group_by(JOUR) %>%
      summarise(total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      mutate(Period = "Reference")
    
    comp <- comp_data() %>%
      group_by(JOUR) %>%
      summarise(total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      mutate(Period = "Comparison")
    
    bind_rows(ref, comp) %>%
      ggplot(aes(x = JOUR, y = total, color = Period)) +
      geom_line(linewidth = 1) +
      scale_y_continuous(labels = comma) +
      labs(title = "Daily Ridership Comparison",
           x = "Date", y = "Total Validations") +
      theme_minimal()
  })
  
  # Station map
  output$station_map <- renderLeaflet({
    leaflet(zd_wgs84) %>%
      addTiles() %>%
      addPolygons(
        group = "stations",
        color = "#4A90E2",
        weight = 1,
        fillOpacity = 0.3,
        popup = ~nom_lda,
        layerId = ~idrefa_lda,
        highlightOptions = highlightOptions(
          weight = 2,
          color = "#FFD700",
          bringToFront = TRUE
        )
      ) %>%
      setView(lng = 2.35, lat = 48.85, zoom = 11)
  })
  
  selected_station_sf <- reactive({
    req(input$station_select)
    
    zd_wgs84 %>%
      dplyr::filter(idrefa_lda == input$station_select)
  })
  
  observe({
    req(selected_station_sf())
    
    bbox <- st_bbox(selected_station_sf())
    
    leafletProxy("station_map") %>%
      fitBounds(
        lng1 = unname(bbox["xmin"]),
        lat1 = unname(bbox["ymin"]),
        lng2 = unname(bbox["xmax"]),
        lat2 = unname(bbox["ymax"])
      )
  })
  
  
  # Station info
  output$station_info <- renderText({
    req(input$station_select)
    
    station_data <- validations %>%
      filter(ID_REFA_LDA == input$station_select)
    
    if(nrow(station_data) == 0) return("No data available for this station")
    
    total_val <- sum(station_data$NB_VALD, na.rm = TRUE)
    avg_daily <- mean(station_data$NB_VALD, na.rm = TRUE)
    station_name <- first(station_data$nom_lda)
    station_type <- first(station_data$type_arret)
    
    paste0(
      "Station: ", station_name, "\n",
      "Type: ", station_type, "\n",
      "Total Validations: ", format(total_val, big.mark = ","), "\n",
      "Avg Daily: ", format(round(avg_daily, 0), big.mark = ","), "\n",
      "Date Range: ", min(station_data$JOUR), " to ", max(station_data$JOUR)
    )
  })
  
  # Station trend plot
  output$station_trend_plot <- renderPlot({
    req(input$station_select)
    
    validations %>%
      filter(ID_REFA_LDA == input$station_select) %>%
      group_by(week = floor_date(JOUR, "week")) %>%
      summarise(total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      ggplot(aes(x = week, y = total)) +
      geom_line(color = "darkblue", linewidth = 1) +
      geom_smooth(method = "loess", color = "red", se = FALSE) +
      scale_y_continuous(labels = comma) +
      labs(title = "Weekly Ridership Trend",
           x = "Week", y = "Total Validations") +
      theme_minimal()
  })
  
  # Station weekday plot
  output$station_weekday_plot <- renderPlot({
    req(input$station_select)
    
    validations %>%
      filter(ID_REFA_LDA == input$station_select) %>%
      group_by(day_of_week) %>%
      summarise(avg_val = mean(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      ggplot(aes(x = day_of_week, y = avg_val, fill = day_of_week)) +
      geom_col(show.legend = FALSE) +
      scale_y_continuous(labels = comma) +
      labs(title = "Average by Weekday",
           x = "Day", y = "Avg Validations") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
  
  # Station category plot
  output$station_category_plot <- renderPlot({
    req(input$station_select)
    
    validations %>%
      filter(ID_REFA_LDA == input$station_select) %>%
      group_by(CATEGORIE_TITRE) %>%
      summarise(total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      filter(!is.na(CATEGORIE_TITRE)) %>%
      ggplot(aes(x = reorder(CATEGORIE_TITRE, -total), y = total, fill = CATEGORIE_TITRE)) +
      geom_col(show.legend = FALSE) +
      scale_y_continuous(labels = comma) +
      labs(title = "Ridership by Pass Category",
           x = "Category", y = "Total Validations") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
  
  # Weekly trend plot
  output$weekly_trend_plot <- renderPlot({
    validations %>%
      group_by(week = floor_date(JOUR, "week")) %>%
      summarise(total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      filter(week != min(week) & week != max(week)) %>%
      ggplot(aes(x = week, y = total)) +
      geom_line(color = "#2C3E50", linewidth = 1) +
      geom_smooth(method = "loess", color = "red", se = FALSE) +
      scale_y_continuous(labels = comma) +
      labs(title = "Weekly Total Validations Over Time",
           x = "Week", y = "Total Validations") +
      theme_minimal()
  })
  
  # Top stations plot
  output$top_stations_plot <- renderPlot({
    validations %>%
      group_by(ID_REFA_LDA, nom_lda) %>%
      summarise(avg_daily = mean(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      slice_max(avg_daily, n = 20) %>%
      ggplot(aes(x = reorder(nom_lda, avg_daily), y = avg_daily)) +
      geom_col(fill = "#3498DB") +
      coord_flip() +
      scale_y_continuous(labels = comma) +
      labs(title = "Top 20 Stations by Avg Daily Ridership",
           x = NULL, y = "Avg Daily Validations") +
      theme_minimal()
  })
  
  # Mode plot
  output$mode_plot <- renderPlot({
    validations %>%
      group_by(type_arret) %>%
      summarise(total = sum(NB_VALD, na.rm = TRUE), .groups = "drop") %>%
      filter(!is.na(type_arret)) %>%
      ggplot(aes(x = reorder(type_arret, -total), y = total, fill = type_arret)) +
      geom_col(show.legend = FALSE) +
      scale_y_continuous(labels = comma) +
      labs(title = "Total Ridership by Transport Mode",
           x = "Mode", y = "Total Validations") +
      theme_minimal() +
      theme(axis.text.x = element_text(angle = 45, hjust = 1))
  })
}

shinyApp(ui = ui, server = server)
